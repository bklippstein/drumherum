def project_class
  classname = project_name.gsub(%r\/(.?)/) { "::#{$1.upcase}" }.gsub(%r(?:^|_)(.)/) { $1.upcase }
  unless %r\A(?:::)?([A-Z]\w*(?:::[A-Z]\w*)*)\z/ =~ classname
    raise NameError, "#{classname.inspect} is not a valid constant name!"
  end
  Object.module_eval("::#{$1}", __FILE__, __LINE__)
end
